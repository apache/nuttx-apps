  # Application info

  nuttx_add_application(

    ${CONFIG_IEEE802154_I8SAK_PROGNAME}

    i8sak_main.c
    STACKSIZE
    ${CONFIG_IEEE802154_I8SAK_STACKSIZE}
    PRIORITY
    ${CONFIG_IEEE802154_I8SAK_PRIORITY})

  set(CSRCS
      i8sak_acceptassoc.c
      i8sak_assoc.c
      i8sak_scan.c
      i8sak_blaster.c
      i8sak_poll.c
      i8sak_sniffer.c
      i8sak_startpan.c
      i8sak_tx.c
      i8sak_get.c
      i8sak_set.c
      i8sak_reset.c
      i8sak_regdump.c
      i8sak_tracedump.c
      i8sak_events.c)

  target_sources(apps PRIVATE ${CSRCS})

endif()
